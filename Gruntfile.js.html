  • 'use strict';
require('marko/node-require').install();
const lasso = require('lasso');
const testPage = require('./test/playground/test-page.marko');

module.exports = (grunt) => {
    require('ebay-grunt-runner')('module-es6')(grunt);
    grunt.registerTask('generate-demo', function() {
        const done = this.async();
        lasso.configure({
            plugins: [
                'i18n-ebay/optimizer/plugin',
                'lasso-less',
                'lasso-autoprefixer',
                'lasso-marko'
            ],
            require: {
                transforms: [
                    'lasso-babel-env'
                ]
            },
            'url-prefix': './',
            outputDir: 'demo',
            fingerprintsEnabled: false,
            minify: true,
            bundlingEnabled: true,
            resolveCssUrls: true,
            cacheProfile: 'development'
        });
        testPage.renderToString({}, (err, string) => {
            if (err) {
                return grunt.log.error(err);
            }
            grunt.file.write('./demo/index.html', string.replace(/"\/static\//g, '"./'));
            done();
        });
    });

    grunt.config.merge({
        copy: {
            demo: {
                cwd: './demo',
                expand: true,
                src: [
                    '**/*'
                ],
                dest: '.docs/demo'
            }
        },
        clean: {
            demo: './demo'
        }
    });

    grunt.registerTask('document', [
        'clean:documentation',
        'docco-plus',
        'copy:coverageReport',
        'generate-demo',
        'copy:demo'
    ]);
};
