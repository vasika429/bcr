const $ = require('jquery');

module.exports = require('marko-widgets').defineComponent({
    template: require('./template.marko'),

    getInitialState(input) {
        return {
            streaming: input.streaming,
            hidden: input.hidden,
            showMsg: false,
            iOS: false,
            numWW: input.options.barcodeWW || 1
        };
    },

    getTemplateData(state) {
        return {
            streaming: state.streaming,
            hidden: state.hidden,
            showMsg: state.showMsg,
            iOS: state.iOS,
            numWW: state.numWW
        };
    },

    stream: null,

    init() {
        const n = navigator;
        n.getUserMedia = (n.getUserMedia || n.webkitGetUserMedia || n.mozGetUserMedia || n.msGetUserMedia);
        if (!n.getUserMedia) {
            this.setState('hidden', true);
        } else {
            const BarcodeReader = require('./barcode-reader');
            const self = this;
            this.BarcodeReader = new BarcodeReader({
                scanCanvasWidth: 1280,
                scanCanvasHeight: 720,
                skipOrientation: true,
                barcodeWW: this.state.numWW,
                onScan: function(barCodes) {
                    if (barCodes.length > 0) {
                        this.emit('valueChange', barCodes[0]);
                        this.stopScanning();
                    }
                }.bind(this)
            });
            this.on('startScanning', this.startScanning.bind(this));
            $('.barcode-reader .dialog__close--icon').on('click', () => {
                this.closeMsgDialog();
            });
        }
    },

    startScanning() {
        this.once('update', () => {
            const constraints = { audio: false, video: { facingMode: { exact: "environment" } } };
            console.log(constraints);
            const wdgt = this;
            navigator.mediaDevices.getUserMedia(constraints)
                .then(function(stream) {
                    console.log("streaming started");
                    console.log(stream);
                    const video = document.querySelector('video');
                    try {
                        video.srcObject = stream;
                        self.stream = stream;
                    } catch (error) {
                        video.src = URL.createObjectURL(stream);
                    }
                    video.onloadedmetadata = function(e) {
                        const video = document.querySelector('video');
                        video.play();
                        wdgt.BarcodeReader.decodeStream(video);
                    };
                    self.timer = setTimeout(wdgt.showInvalidMsg, 40000, wdgt);
                })
                .catch(function(err) {
                    if (err.name === "NotAllowedError") {
                        if (wdgt) {
                            const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) && !window.MSStream;
                            wdgt.setState('iOS', isIOS);
                            wdgt.setState('streaming', false);
                            wdgt.setState('showMsg', true);
                            setTimeout(() => {
                                wdgt.getWidget('msg_dialog').show();
                            }, 100);
                        }
                    }
                });
        });
        this.setState('streaming', true);
    },

    closeMsgDialog() {
        this.setState('streaming', false);
        this.setState('showMsg', false);
        setTimeout(() => {
            this.getWidget('msg_dialog').hide();
        }, 0);
    },

    stopScanning() {
        this.setState('streaming', false);
        if (self && self.stream && self.stream.getTracks()[0]) {
            self.stream.getTracks()[0].stop();
        }
        this.BarcodeReader.stopStreamDecode();
    },

    showInvalidMsg(obj) {
        $( '.invalid' ).show();
        $( '.background' ).hide();
        clearTimeout(this.timer);
        obj.emit('valueChange', "Barcode timed out");
    }
});
