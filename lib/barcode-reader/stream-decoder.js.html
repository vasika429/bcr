const DEFAULT_FORMATS = [

    'Code93',
    'Code39',
    'EAN-13',
    '2Of5',
    'Inter2Of5',
    'Codabar'

  • The callback function for stream decoding used internally by this.

        barcodeReaderStreamCallback({ data } = {}) {
        if (data) {
            this.emit(data.success ? 'scanSuccess' : 'scanFail', data);
        }
        if (this.videoStream) {
            this.emit('startScan');
            this.scanContext.drawImage(this.videoStream, 0, 0, this.scanCanvas.width, this.scanCanvas.height);
            const imageData = this.scanContext.getImageData(
                0,
                this.scanCanvas.height * cropFactor / 2,
                this.scanCanvas.width,
                this.scanCanvas.height * (1 - cropFactor / 2)
            ).data;
            this.emit('imageData', this.scanCanvas.toDataURL());
            this.decoder.postMessage({
                scan: imageData,
                scanWidth: this.scanCanvas.width,
                scanHeight: this.scanCanvas.height * (1 - cropFactor),
                decodeFormats: DEFAULT_FORMATS
            });
        }
    }
}
