            }

    function otsu(histogramVal, total) {
        let i;
        let sum = 0;
        for (i = 1; i < histogramVal.length; ++i) {
            sum += i * histogramVal[i];
        }
        let sumB = 0;

        let max = 0.0;

        let threshold2 = 0.0;
        for (i = 0; i < histogramVal.length; ++i) {
            wB += histogramVal[i];
            if (wB === 0) {continue;}
            wF = total - wB;
            if (wF === 0) {break;}
            sumB += i * histogramVal[i];
            mB = sumB / wB;
            mF = (sum - sumB) / wF;
