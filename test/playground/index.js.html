  • 'use strict';

require('marko/node-require').install();

const express = require('express');
const serveStatic = require('serve-static');
const lasso = require('lasso');
const testPage = require('./test-page.marko');

lasso.configure({
    plugins: [
        'i18n-ebay/optimizer/plugin',
        'lasso-less',
        'lasso-autoprefixer',
        'lasso-marko'
    ],
    require: {
        transforms: [
            'lasso-babel-env'
        ]
    },
    'url-prefix': '/static',
    outputDir: 'static',
    fingerprintsEnabled: false,
    minify: false,
    bundlingEnabled: false,
    resolveCssUrls: true,
    cacheProfile: 'development'
});

const app = express();

app.use('/static', serveStatic('static'));
app.get('/barcode-reader', (req, res) => {
    testPage.render({}, res);
});

app.listen(process.env.PORT || 8080, () => {
    if (process.send) {
        process.send('online');
    }
});

module.exports = app;
